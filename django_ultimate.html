What is a Django App?

Core

A project is the whole site. An app is a self-contained Python package that owns one domain — students, attendance, payments. You can have many apps in one project, and apps can be reused across projects or published as pip packages.

anatomy of an app directoryFilesystem
students/
├── __init__.py       # makes it a Python package
├── apps.py           # app config class — name, label, ready() for signals
├── models.py         # ★ database tables (class → table)
├── views.py          # ★ request handlers (FBV or CBV)
├── urls.py           # ★ URL patterns for this domain
├── admin.py          # ★ register models in Django admin panel
├── serializers.py    # DRF — convert models ↔ JSON
├── permissions.py    # DRF — custom access rules
├── filters.py        # django-filter FilterSet classes
├── signals.py        # post_save / pre_delete hooks
├── forms.py          # Django Form / ModelForm classes
├── tests.py          # unit + integration tests
├── migrations/       # auto-generated DB migration files — never edit
│   └── 0001_initial.py
└── templates/students/  # HTML templates namespaced under app name
apps.py + settings registrationPython

URL Routing

Core

URL confs map HTTP paths to view callables. The root urls.py delegates to app-level confs via include(). URL names + namespaces let you reverse URLs safely everywhere.

students/urls.py — all pattern typesPython
from django.urls import path, re_path, include
from rest_framework.routers import DefaultRouter
from . import views

router = DefaultRouter()
router.register(r'students', views.StudentViewSet, basename='student')

urlpatterns = [
    path('', include(router.urls)),          # GET/POST /students/ + GET/PUT/DELETE /students/{pk}/

    # int, str, slug, uuid, path converters
    path('<int:pk>/report/',        views.ReportView.as_view()),
    path('<slug:roll_no>/profile/',  views.ProfileView.as_view()),
    path('<uuid:student_uuid>/',     views.StudentByUUID.as_view()),

    # regex for complex patterns
    re_path(r'^batch/(?P<year>\d{4})/$', views.BatchView.as_view()),
]

# Reverse: reverse('students:student-list')
# Template: {% url 'students:student-detail' pk=s.pk %}

Views — FBV & CBV

Core

Any callable that takes HttpRequest and returns HttpResponse is a view. Functions are simple and explicit; class-based views enable inheritance and mixins.

FBV vs CBV side by sidePython
# ── Function-Based View ───────────────────────────────────────
from django.shortcuts import render, get_object_or_404, redirect
from django.contrib.auth.decorators import login_required

@login_required
def student_detail(request, pk):
    student = get_object_or_404(Student, pk=pk)
    # request.method / .user / .POST / .FILES / .GET / .session
    return render(request, 'students/detail.html', {'student': student})


# ── Class-Based View ──────────────────────────────────────────
from django.views.generic import ListView, DetailView, CreateView
from django.contrib.auth.mixins import LoginRequiredMixin

class StudentListView(LoginRequiredMixin, ListView):
    model               = Student
    template_name       = 'students/list.html'
    context_object_name = 'students'
    paginate_by         = 25

    def get_queryset(self):
        return Student.objects.filter(is_active=True).select_related('classroom')

    def get_context_data(self, **kwargs):
        ctx = super().get_context_data(**kwargs)
        ctx['total'] = self.get_queryset().count()
        return ctx

Models & ORM

Core

Models are Python classes mapping to database tables. The ORM translates method calls to SQL — no raw SQL needed. Field choices, Meta options, relationships, and query patterns are all shown below.

models.py — all field types + relationships + ORM patternsPython
from django.db import models
from django.conf import settings

class BaseModel(models.Model):
    """Abstract base — gives every model created_at, updated_at."""
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    class Meta: abstract = True


class Student(BaseModel):
    # String fields
    full_name  = models.CharField(max_length=150)
    roll_no    = models.SlugField(unique=True)
    email      = models.EmailField(unique=True)
    address    = models.TextField(blank=True)

    # Numeric
    age        = models.PositiveSmallIntegerField()
    gpa        = models.DecimalField(max_digits=4, decimal_places=2, default=0)

    # Choice field
    GENDER     = [('M','Male'),('F','Female'),('O','Other')]
    gender     = models.CharField(max_length=1, choices=GENDER)

    # Boolean & date
    is_active  = models.BooleanField(default=True)
    dob        = models.DateField()
    joined_on  = models.DateField(auto_now_add=True)

    # File / image
    photo      = models.ImageField(upload_to='students/%Y/%m/', null=True, blank=True)
    documents  = models.JSONField(default=list, blank=True)

    # Relationships
    classroom  = models.ForeignKey('classes.Classroom', on_delete=models.SET_NULL,
                                   null=True, related_name='students')
    guardian   = models.OneToOneField('Guardian', on_delete=models.SET_NULL,
                                      null=True, related_name='ward')
    subjects   = models.ManyToManyField('teachers.Subject', blank=True)

    class Meta:
        ordering       = ['full_name']
        indexes        = [models.Index(fields=['roll_no'], name='student_roll_idx')]
        unique_together = [('full_name', 'dob')]
        verbose_name   = 'Student';  verbose_name_plural = 'Students'

    def __str__(self): return f"{self.full_name} ({self.roll_no})"

Forms & ModelForms

Core

Django Forms handle HTML rendering, parsing POST data, validation (per-field and cross-field), and returning cleaned data or error messages. ModelForm auto-generates fields from a model — preferred 90% of the time.

forms.py — ModelForm with custom validationPython
from django import forms
from .models import Student

class StudentForm(forms.ModelForm):
    class Meta:
        model  = Student
        fields = ['full_name', 'roll_no', 'email', 'dob', 'gender', 'photo']
        widgets = {'dob': forms.DateInput(attrs={'type':'date'})}

    def clean_roll_no(self):             # field-level
        roll = self.cleaned_data['roll_no']
        if not roll.startswith('2024'):
            raise forms.ValidationError('Roll number must start with the current year.')
        return roll.upper()

    def clean(self):                     # cross-field
        data   = super().clean()
        name   = data.get('full_name', '')
        email  = data.get('email', '')
        if name.lower() in email.lower():
            self.add_error('email', 'Do not use your name in the email address.')
        return data

Custom Middleware

Middleware

Write custom middleware when you need global logic that applies to every request — request timing, maintenance mode, tenant detection, IP blocking, audit logging.

core/middleware.py — three real examplesPython
import time, logging
from django.http import HttpResponse

logger = logging.getLogger('school.requests')

# ── Example 1: Request timing (function-style) ────────────────────
def RequestLoggingMiddleware(get_response):
    def middleware(request):
        start    = time.perf_counter()
        response = get_response(request)   # ← view runs here
        duration = (time.perf_counter() - start) * 1000
        response['X-Response-Time'] = ff"{duration:.1f}ms"
        logger.info(ff"{request.method} {request.path} {response.status_code} {duration:.0f}ms")
        return response
    return middleware


# ── Example 2: Maintenance mode (class-style with all hooks) ─────────
class MaintenanceModeMiddleware:
    EXEMPT = ['/admin/', '/api/auth/token/', '/health/']

    def __init__(self, get_response):
        self.get_response = get_response   # runs once at startup

    def __call__(self, request):
        # ── REQUEST PHASE ──────────────────────────────────────────
        from django.conf import settings
        if getattr(settings, 'MAINTENANCE_MODE', False):
            exempt = any(request.path.startswith(p) for p in self.EXEMPT)
            if not exempt and not request.user.is_staff:
                return HttpResponse('{"error":"Maintenance mode"}',
                                    status=503, content_type='application/json')

        response = self.get_response(request)   # ← view runs here
        # ── RESPONSE PHASE ─────────────────────────────────────────
        response['X-School-API'] = 'v1'
        return response

    def process_view(self, request, view_func, args, kwargs):
        # Called just BEFORE the view — return None to continue
        pass

    def process_exception(self, request, exception):
        logger.error(ff"Exception: {exception} on {request.path}", exc_info=True)


# ── Example 3: Role-based access for specific URL prefix ─────────────
def AdminAreaMiddleware(get_response):
    def middleware(request):
        if request.path.startswith('/api/admin-reports/'):
            if not (request.user.is_authenticated and request.user.is_staff):
                return HttpResponse('{"error":"Staff only"}', status=403,
                                    content_type='application/json')
        return get_response(request)
    return middleware

Serializers

DRF

A Serializer converts Python objects → JSON (serialization) and JSON → validated Python objects (deserialization). ModelSerializer auto-generates fields from a model with one-liner Meta.

serializers.py — all patternsPython
from rest_framework import serializers
from .models import Student

class StudentSerializer(serializers.ModelSerializer):
    # Computed (read-only)
    age_display  = serializers.SerializerMethodField()
    # Traverse FK — read author.full_name as string
    classroom_name = serializers.CharField(source='classroom.name', read_only=True)
    # Write-only field (e.g. password)
    secret_code  = serializers.CharField(write_only=True, required=False)
    # Hyperlink to related object
    photo_url    = serializers.SerializerMethodField()

    class Meta:
        model  = Student
        fields = ['id', 'roll_no', 'full_name', 'email', 'gpa',
                  'age_display', 'classroom_name', 'photo_url',
                  'secret_code', 'created_at']
        read_only_fields = ['id', 'created_at', 'gpa']
        extra_kwargs = {
            'email': {'validators': []},   # remove unique check for update
            'full_name': {'min_length': 3},
        }

    def get_age_display(self, obj):
        from datetime import date
        today = date.today()
        age   = today.year - obj.dob.year - ((today.month, today.day) < (obj.dob.month, obj.dob.day))
        return ff"{age} years"

    def get_photo_url(self, obj):
        request = self.context.get('request')
        if obj.photo and request:
            return request.build_absolute_uri(obj.photo.url)
        return None

    def create(self, validated_data):
        validated_data.pop('secret_code', None)   # remove before save
        return super().create(validated_data)

from rest_framework import serializers
from rest_framework.validators import UniqueValidator, UniqueTogetherValidator

def valid_roll_format(value):           # Level 1: standalone validator function
    if not value[:4].isdigit():
        raise serializers.ValidationError('Roll number must start with 4 digits (year).')

class StudentSerializer(serializers.ModelSerializer):
    # Level 2: field-level validators list
    roll_no = serializers.SlugField(validators=[
        valid_roll_format,
        UniqueValidator(queryset=Student.objects.all(), message='Roll number taken.')
    ])

    class Meta:
        model  = Student
        fields = ['roll_no', 'full_name', 'email', 'dob', 'classroom']
        validators = [
            UniqueTogetherValidator(queryset=Student.objects.all(),
                                    fields=['full_name','dob'],
                                    message='Student already enrolled.')
        ]

    # Level 3: validate_<fieldname> — field-level, must RETURN value
    def validate_full_name(self, value):
        if len(value.strip().split()) < 2:
            raise serializers.ValidationError('Please provide first and last name.')
        return value.strip().title()

    # Level 4: validate() — cross-field, must RETURN attrs
    def validate(self, attrs):
        from datetime import date
        dob       = attrs.get('dob')
        classroom = attrs.get('classroom')
        if dob and classroom:
            age = date.today().year - dob.year
            if abs(age - classroom.expected_age) > 3:
                raise serializers.ValidationError(
                    {'dob': ff"Age {age} doesn't match classroom grade level."}
                )
        return attrs

class StudentListAPIView(APIView):
    permission_classes = [permissions.IsAuthenticatedOrReadOnly]

Permissions

DRF

Permissions answer: "is this user allowed to do this?" All classes in permission_classes must pass (AND). Combine with | for OR. has_permission runs for every request; has_object_permission runs only on detail views.

Custom permissions + per-actionPython
from rest_framework import permissions

# ── Built-ins ─────────────────────────────────────────────────────────
# AllowAny, IsAuthenticated, IsAdminUser, IsAuthenticatedOrReadOnly
# DjangoModelPermissions, DjangoObjectPermissions

# ── Custom: teacher can edit their own students only ──────────────────
class IsTeacherOrAdmin(permissions.BasePermission):
    message = 'Only teachers or admins can access this.'

    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS: return True
        return request.user.is_authenticated and (
            request.user.role in ['teacher', 'admin'] or request.user.is_staff
        )

    def has_object_permission(self, request, view, obj):
        if request.method in permissions.SAFE_METHODS: return True
        if request.user.is_staff: return True
        return getattr(obj, 'classroom', None) and \
               obj.classroom.teacher == request.user


class IsOwnerStudent(permissions.BasePermission):
    """Student can only see/edit their own record."""
    def has_object_permission(self, request, view, obj):
        return request.user.is_staff or obj.user == request.user


# ── Per-action permissions in ViewSet ─────────────────────────────────
class StudentViewSet(viewsets.ModelViewSet):
    def get_permissions(self):
        if self.action in ['list', 'retrieve']:
            return [permissions.IsAuthenticated()]
        if self.action in ['create', 'update', 'partial_update']:
            return [permissions.IsAuthenticated(), IsTeacherOrAdmin()]
        if self.action == 'destroy':
            return [permissions.IsAdminUser()]
        return [permissions.IsAuthenticated()]

logger = logging.getLogger('school.api')


def custom_exception_handler(exc, context):
    """
    Consistent error shape for ALL API errors:
    {"success": false, "code": 404, "message": "...", "errors": {...}}
    """
    response = exception_handler(exc, context)   # let DRF handle first

    if response is not None:
        logger.warning('api_error', extra={
            'status': response.status_code,
            'exc':    exc.__class__.__name__,
            'path':   context['request'].path,
        })
        response.data = {
            'success': False,
            'code':    response.status_code,
            'message': str(exc.detail) if hasattr(exc, 'detail') else str(exc),
            'errors':  response.data,
        }
    return response

            'student': StudentListSerializer(student, context={'request': request}).data,

        })

        student = self.get_object()

        total   = student.attendance.count()
        present = student.attendance.filter(status='P').count()
        pct     = round((present / total * 100), 1) if total else 0

    message = 'Only teachers or admins can perform this action.'
    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS:
            return request.user.is_authenticated
        return (request.user.is_authenticated and
                request.user.role in ['teacher', 'admin'] or
                request.user.is_staff)

        if request.method in permissions.SAFE_METHODS: return True
        if request.user.is_staff or request.user.role == 'admin': return True
        # Teacher can only modify students in their classroom
