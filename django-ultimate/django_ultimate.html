What is a Django App?

Core

A project is the whole site. An app is a self-contained Python package that owns one domain — students, attendance, payments. You can have many apps in one project, and apps can be reused across projects or published as pip packages.

anatomy of an app directoryFilesystem
students/
├── __init__.py       # makes it a Python package
├── apps.py           # app config class — name, label, ready() for signals
├── models.py         # ★ database tables (class → table)
├── views.py          # ★ request handlers (FBV or CBV)
├── urls.py           # ★ URL patterns for this domain
├── admin.py          # ★ register models in Django admin panel
├── serializers.py    # DRF — convert models ↔ JSON
├── permissions.py    # DRF — custom access rules
├── filters.py        # django-filter FilterSet classes
├── signals.py        # post_save / pre_delete hooks
├── forms.py          # Django Form / ModelForm classes
├── tests.py          # unit + integration tests
├── migrations/       # auto-generated DB migration files — never edit
│   └── 0001_initial.py
└── templates/students/  # HTML templates namespaced under app name
apps.py + settings registrationPython
# students/apps.py

class StudentsConfig(AppConfig):
    name             = 'students'

        import students.signals  # noqa — wire up signal handlers at startup

# config/settings.py
INSTALLED_APPS = [
    # ...django built-ins...

URL Routing

Core

URL confs map HTTP paths to view callables. The root urls.py delegates to app-level confs via include(). URL names + namespaces let you reverse URLs safely everywhere.

students/urls.py — all pattern typesPython
from django.urls import path, re_path, include
from rest_framework.routers import DefaultRouter
from . import views

router = DefaultRouter()
router.register(r'students', views.StudentViewSet, basename='student')

urlpatterns = [
    path('', include(router.urls)),          # GET/POST /students/ + GET/PUT/DELETE /students/{pk}/

    # int, str, slug, uuid, path converters
    path('<int:pk>/report/',        views.ReportView.as_view()),
    path('<slug:roll_no>/profile/',  views.ProfileView.as_view()),
    path('<uuid:student_uuid>/',     views.StudentByUUID.as_view()),

    # regex for complex patterns
    re_path(r'^batch/(?P<year>\d{4})/$', views.BatchView.as_view()),
]

# Reverse: reverse('students:student-list')
# Template: {% url 'students:student-detail' pk=s.pk %}

Views — FBV & CBV

Core

Any callable that takes HttpRequest and returns HttpResponse is a view. Functions are simple and explicit; class-based views enable inheritance and mixins.

FBV vs CBV side by sidePython
# ── Function-Based View ───────────────────────────────────────
from django.shortcuts import render, get_object_or_404, redirect
from django.contrib.auth.decorators import login_required

@login_required
def student_detail(request, pk):
    student = get_object_or_404(Student, pk=pk)
    # request.method / .user / .POST / .FILES / .GET / .session
    return render(request, 'students/detail.html', {'student': student})


# ── Class-Based View ──────────────────────────────────────────
from django.views.generic import ListView, DetailView, CreateView
from django.contrib.auth.mixins import LoginRequiredMixin

class StudentListView(LoginRequiredMixin, ListView):
    model               = Student
    template_name       = 'students/list.html'
    context_object_name = 'students'
    paginate_by         = 25

    def get_queryset(self):
        return Student.objects.filter(is_active=True).select_related('classroom')

    def get_context_data(self, **kwargs):
        ctx = super().get_context_data(**kwargs)
        ctx['total'] = self.get_queryset().count()
        return ctx

Models & ORM

Core

Models are Python classes mapping to database tables. The ORM translates method calls to SQL — no raw SQL needed. Field choices, Meta options, relationships, and query patterns are all shown below.

models.py — all field types + relationships + ORM patternsPython
from django.db import models
from django.conf import settings

class BaseModel(models.Model):
    """Abstract base — gives every model created_at, updated_at."""
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    class Meta: abstract = True


class Student(BaseModel):
    # String fields
    full_name  = models.CharField(max_length=150)
    roll_no    = models.SlugField(unique=True)
    email      = models.EmailField(unique=True)
    address    = models.TextField(blank=True)

    # Numeric
    age        = models.PositiveSmallIntegerField()
    gpa        = models.DecimalField(max_digits=4, decimal_places=2, default=0)

    # Choice field
    GENDER     = [('M','Male'),('F','Female'),('O','Other')]
    gender     = models.CharField(max_length=1, choices=GENDER)

    # Boolean & date
    is_active  = models.BooleanField(default=True)
    dob        = models.DateField()
    joined_on  = models.DateField(auto_now_add=True)

    # File / image
    photo      = models.ImageField(upload_to='students/%Y/%m/', null=True, blank=True)
    documents  = models.JSONField(default=list, blank=True)

    # Relationships
    classroom  = models.ForeignKey('classes.Classroom', on_delete=models.SET_NULL,
                                   null=True, related_name='students')
    guardian   = models.OneToOneField('Guardian', on_delete=models.SET_NULL,
                                      null=True, related_name='ward')
    subjects   = models.ManyToManyField('teachers.Subject', blank=True)

    class Meta:
        ordering       = ['full_name']
        indexes        = [models.Index(fields=['roll_no'], name='student_roll_idx')]
        unique_together = [('full_name', 'dob')]
        verbose_name   = 'Student';  verbose_name_plural = 'Students'

    def __str__(self): return f"{self.full_name} ({self.roll_no})"


# ── ORM query patterns ─────────────────────────────────────────────

ORM Relationships — One-to-One · One-to-Many · Many-to-One · Many-to-Many

Core

Django provides three relationship field types that map directly to SQL foreign-key patterns. Each one is explained below with model definitions, the DB tables Django creates, and ORM queries from both sides of the relationship — the side that holds the field and the reverse accessor side.

① One-to-One  OneToOneField

Each row in Table A maps to exactly one row in Table B — and vice versa. Django adds a UNIQUE constraint on the FK column so no two A rows can point to the same B row. Classic use-case: extending an existing model without subclassing (e.g. a Profile record per User, or a Guardian per Student).

models.py — OneToOneFieldPython
from django.db import models

class Guardian(models.Model):
    full_name = models.CharField(max_length=150)
    phone     = models.CharField(max_length=15)
    relation  = models.CharField(max_length=30)   # 'Father', 'Mother', etc.

    def __str__(self): return self.full_name


class Student(models.Model):
    full_name = models.CharField(max_length=150)
    roll_no   = models.SlugField(unique=True)

    # OneToOneField — each student has ONE guardian, each guardian belongs to ONE student
    guardian  = models.OneToOneField(
        Guardian,
        on_delete=models.SET_NULL,   # if guardian deleted, set student.guardian = NULL
        null=True,
        blank=True,
        related_name='ward',        # reverse accessor: guardian_obj.ward → Student
    )

    def __str__(self): return self.full_name
DB TableColumn addedConstraint
students_studentguardian_id (FK → students_guardian.id)UNIQUE — enforces the "one" on both sides
students_guardian—No change; reverse access is via Python only
ORM queries — forward & reversePython
# ── Forward: student → guardian ────────────────────────────────────
s = Student.objects.select_related('guardian').get(roll_no='2024001')
s.guardian.full_name       # 'John Doe'  — one JOIN, no extra query
s.guardian.phone           # '9876543210'

# Set / change guardian
g = Guardian.objects.create(full_name='Jane Doe', phone='9876543211', relation='Mother')
s.guardian = g
s.save()

# ── Reverse: guardian → student ────────────────────────────────────
g = Guardian.objects.get(pk=1)
g.ward            # Student object  ← related_name='ward'
g.ward.full_name  # 'Alice'

# Check if guardian has a ward
try:
    ward = g.ward
except Student.DoesNotExist:
    ward = None

# OR use hasattr
if hasattr(g, 'ward'): print(g.ward)

② One-to-Many  ForeignKey — forward (many → one)

Many rows in Table A reference one row in Table B. This is Django's ForeignKey. The FK column lives on the "many" side. Example: many Student rows share one Classroom. From the student's perspective this is a many-to-one lookup; from the classroom's perspective it is one-to-many.

models.py — ForeignKeyPython
class Classroom(models.Model):
    name     = models.CharField(max_length=20)   # e.g. '10A'
    capacity = models.PositiveSmallIntegerField(default=40)

    def __str__(self): return self.name


class Student(models.Model):
    full_name = models.CharField(max_length=150)

    # ForeignKey — many students belong to ONE classroom
    classroom = models.ForeignKey(
        Classroom,
        on_delete=models.SET_NULL,   # options: CASCADE, PROTECT, SET_NULL, DO_NOTHING
        null=True,
        blank=True,
        related_name='students',    # reverse accessor: classroom.students.all()
    )
on_delete optionWhat happens when the Classroom is deleted
CASCADEDelete all related students too
PROTECTRaise ProtectedError — prevents deletion while students exist
SET_NULLSet student.classroom = NULL (requires null=True)
SET_DEFAULTSet the FK to the field's default value
DO_NOTHINGLeave DB as-is — you handle integrity manually
ORM queries — forward (many → one) and reverse (one → many)Python
# ── Forward: student → classroom (many-to-one) ─────────────────────
s = Student.objects.select_related('classroom').get(pk=1)
s.classroom           # Classroom object
s.classroom.name      # '10A'
s.classroom_id        # raw FK integer — no extra DB hit

# Filter students that belong to classroom '10A'
Student.objects.filter(classroom__name='10A')

# Assign a student to a different classroom
room = Classroom.objects.get(name='11B')
s.classroom = room
s.save()

# ── Reverse: classroom → students (one-to-many) ────────────────────
room = Classroom.objects.prefetch_related('students').get(name='10A')
room.students.all()            # QuerySet of all students in 10A
room.students.count()          # 35
room.students.filter(gpa__gte=8)

# Annotate classrooms with student count
from django.db.models import Count
Classroom.objects.annotate(student_count=Count('students'))

# Add a new student directly via reverse manager
room.students.create(full_name='New Student')

③ Many-to-One — the same ForeignKey, the other perspective

Many-to-one is not a separate field type — it is how a ForeignKey looks from the "many" side. When a student accesses student.classroom that is a many-to-one traversal: many students share one classroom. The table below shows the same relationship from both perspectives to eliminate confusion.

DirectionTermExampleORM accessor
Child → ParentMany-to-Onestudent → classroomstudent.classroom (returns one object)
Parent → ChildrenOne-to-Manyclassroom → studentsclassroom.students.all() (returns QuerySet)
Real-world many-to-one patternsPython
# Pattern 1 — get every student's classroom name in one query (no N+1)
students = Student.objects.select_related('classroom').all()
for s in students:
    print(s.full_name, '→', s.classroom.name)   # uses JOIN, not per-row query

# Pattern 2 — span across the FK with double-underscore lookups
Student.objects.filter(classroom__capacity__gte=40)
Student.objects.filter(classroom__name__startswith='10')
Student.objects.values('classroom__name', 'full_name')

# Pattern 3 — group students by classroom
from django.db.models import Count
Classroom.objects.annotate(total=Count('students')).order_by('-total')

# Pattern 4 — move all students from one class to another (bulk update)
old_room = Classroom.objects.get(name='10A')
new_room = Classroom.objects.get(name='10B')
Student.objects.filter(classroom=old_room).update(classroom=new_room)

④ Many-to-Many  ManyToManyField

Many rows in Table A relate to many rows in Table B. Django automatically creates a junction / through table with two FK columns to manage the relationship. Classic use-case: a student enrolls in many subjects, and each subject is taken by many students. You can also define a custom through model to store extra data on the relationship (e.g. enrollment date, grade).

models.py — ManyToManyField (auto junction table)Python
class Subject(models.Model):
    name = models.CharField(max_length=100)
    code = models.CharField(max_length=10, unique=True)

    def __str__(self): return self.name


class Student(models.Model):
    full_name = models.CharField(max_length=150)

    # ManyToManyField — a student takes many subjects, a subject has many students
    subjects  = models.ManyToManyField(
        Subject,
        blank=True,
        related_name='enrolled_students',  # subject.enrolled_students.all()
    )
DB TableColumnsNote
students_studentno new columnM2M fields don't add a column to the model's table
students_subjectno new columnSame
students_student_subjects (auto)student_id, subject_idDjango creates this junction table automatically
ORM queries — M2M add/remove/set + reversePython
# ── Student → Subjects (forward) ───────────────────────────────────
s = Student.objects.prefetch_related('subjects').get(pk=1)

s.subjects.all()                          # all subjects this student takes
s.subjects.count()                        # 5
s.subjects.filter(code__startswith='SCI')

# Add a subject (insert row in junction table)
maths = Subject.objects.get(code='MATH01')
s.subjects.add(maths)

# Add multiple at once
s.subjects.add(*Subject.objects.filter(code__in=['SCI01', 'ENG01']))

# Remove a subject
s.subjects.remove(maths)

# Replace ALL subjects at once
s.subjects.set(Subject.objects.filter(code__startswith='CORE'))

# Remove all subjects
s.subjects.clear()

# ── Subject → Students (reverse) ───────────────────────────────────
subj = Subject.objects.get(code='MATH01')
subj.enrolled_students.all()             # all students taking maths
subj.enrolled_students.filter(gpa__gte=9)

# ── Cross-filter (span the M2M) ────────────────────────────────────
# students who take both MATH01 AND SCI01
Student.objects.filter(subjects__code='MATH01').filter(subjects__code='SCI01')

# subjects taken by student with roll 2024001
Subject.objects.filter(enrolled_students__roll_no='2024001')

④b Many-to-Many with a custom through model

When you need to store extra data on the relationship itself (e.g. the date a student enrolled in a subject, or the grade they received), replace the auto junction table with an explicit through model.

models.py — explicit through model (Enrollment)Python
class Enrollment(models.Model):
    # The junction table — with extra fields
    student      = models.ForeignKey(Student, on_delete=models.CASCADE, related_name='enrollments')
    subject      = models.ForeignKey(Subject, on_delete=models.CASCADE, related_name='enrollments')
    enrolled_on  = models.DateField(auto_now_add=True)

import time, logging
from django.http import HttpResponse

logger = logging.getLogger('school.requests')

# ── Example 1: Request timing (function-style) ────────────────────
def RequestLoggingMiddleware(get_response):
    def middleware(request):
        start    = time.perf_counter()
        response = get_response(request)   # ← view runs here
        duration = (time.perf_counter() - start) * 1000
        response['X-Response-Time'] = ff"{duration:.1f}ms"
        logger.info(ff"{request.method} {request.path} {response.status_code} {duration:.0f}ms")
        return response
    return middleware


# ── Example 2: Maintenance mode (class-style with all hooks) ─────────
class MaintenanceModeMiddleware:
    EXEMPT = ['/admin/', '/api/auth/token/', '/health/']

    def __init__(self, get_response):
        self.get_response = get_response   # runs once at startup

    def __call__(self, request):
        # ── REQUEST PHASE ──────────────────────────────────────────
        from django.conf import settings
        if getattr(settings, 'MAINTENANCE_MODE', False):
            exempt = any(request.path.startswith(p) for p in self.EXEMPT)
            if not exempt and not request.user.is_staff:
                return HttpResponse('{"error":"Maintenance mode"}',
                                    status=503, content_type='application/json')

        response = self.get_response(request)   # ← view runs here
        # ── RESPONSE PHASE ─────────────────────────────────────────
        response['X-School-API'] = 'v1'
        return response

    def process_view(self, request, view_func, args, kwargs):
        # Called just BEFORE the view — return None to continue
        pass

    def process_exception(self, request, exception):
        logger.error(ff"Exception: {exception} on {request.path}", exc_info=True)


# ── Example 3: Role-based access for specific URL prefix ─────────────
def AdminAreaMiddleware(get_response):
    def middleware(request):
        if request.path.startswith('/api/admin-reports/'):
            if not (request.user.is_authenticated and request.user.is_staff):
                return HttpResponse('{"error":"Staff only"}', status=403,
                                    content_type='application/json')

    return middleware

from rest_framework import serializers
from .models import Student

class StudentSerializer(serializers.ModelSerializer):
    # Computed (read-only)
    age_display  = serializers.SerializerMethodField()
    # Traverse FK — read author.full_name as string
    classroom_name = serializers.CharField(source='classroom.name', read_only=True)
    # Write-only field (e.g. password)
    secret_code  = serializers.CharField(write_only=True, required=False)
    # Hyperlink to related object
    photo_url    = serializers.SerializerMethodField()

    class Meta:
        model  = Student
        fields = ['id', 'roll_no', 'full_name', 'email', 'gpa',
                  'age_display', 'classroom_name', 'photo_url',
                  'secret_code', 'created_at']
        read_only_fields = ['id', 'created_at', 'gpa']
        extra_kwargs = {
            'email': {'validators': []},   # remove unique check for update
            'full_name': {'min_length': 3},
        }

    def get_age_display(self, obj):
        from datetime import date
        today = date.today()
        age   = today.year - obj.dob.year - ((today.month, today.day) < (obj.dob.month, obj.dob.day))
        return ff"{age} years"

    def get_photo_url(self, obj):
        request = self.context.get('request')
        if obj.photo and request:
            return request.build_absolute_uri(obj.photo.url)
        return None

    def create(self, validated_data):
        validated_data.pop('secret_code', None)   # remove before save
        return super().create(validated_data)

from rest_framework import serializers
from rest_framework.validators import UniqueValidator, UniqueTogetherValidator

def valid_roll_format(value):           # Level 1: standalone validator function
    if not value[:4].isdigit():
        raise serializers.ValidationError('Roll number must start with 4 digits (year).')

class StudentSerializer(serializers.ModelSerializer):
    # Level 2: field-level validators list
    roll_no = serializers.SlugField(validators=[
        valid_roll_format,
        UniqueValidator(queryset=Student.objects.all(), message='Roll number taken.')
    ])

    class Meta:
        model  = Student
        fields = ['roll_no', 'full_name', 'email', 'dob', 'classroom']
        validators = [
            UniqueTogetherValidator(queryset=Student.objects.all(),
                                    fields=['full_name','dob'],
                                    message='Student already enrolled.')
        ]

    # Level 3: validate_<fieldname> — field-level, must RETURN value
    def validate_full_name(self, value):
        if len(value.strip().split()) < 2:
            raise serializers.ValidationError('Please provide first and last name.')
        return value.strip().title()

    # Level 4: validate() — cross-field, must RETURN attrs
    def validate(self, attrs):
        from datetime import date
        dob       = attrs.get('dob')
        classroom = attrs.get('classroom')
        if dob and classroom:
            age = date.today().year - dob.year
            if abs(age - classroom.expected_age) > 3:
                raise serializers.ValidationError(
                    {'dob': ff"Age {age} doesn't match classroom grade level."}
                )
        return attrs

class StudentViewSet(viewsets.ModelViewSet):

Permissions answer: "is this user allowed to do this?" All classes in permission_classes must pass (AND). Combine with | for OR. has_permission runs for every request; has_object_permission runs only on detail views.

from rest_framework import permissions

# AllowAny, IsAuthenticated, IsAdminUser, IsAuthenticatedOrReadOnly
# DjangoModelPermissions, DjangoObjectPermissions

# ── Custom: teacher can edit their own students only ──────────────────
class IsTeacherOrAdmin(permissions.BasePermission):
    message = 'Only teachers or admins can access this.'

    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS: return True
        return request.user.is_authenticated and (
            request.user.role in ['teacher', 'admin'] or request.user.is_staff
        )

    def has_object_permission(self, request, view, obj):
        if request.method in permissions.SAFE_METHODS: return True
        if request.user.is_staff: return True
        return getattr(obj, 'classroom', None) and \
               obj.classroom.teacher == request.user


class IsOwnerStudent(permissions.BasePermission):
    """Student can only see/edit their own record."""
    def has_object_permission(self, request, view, obj):
        return request.user.is_staff or obj.user == request.user


# ── Per-action permissions in ViewSet ─────────────────────────────────
class StudentViewSet(viewsets.ModelViewSet):
    def get_permissions(self):
        if self.action in ['list', 'retrieve']:
            return [permissions.IsAuthenticated()]
        if self.action in ['create', 'update', 'partial_update']:
            return [permissions.IsAuthenticated(), IsTeacherOrAdmin()]
        if self.action == 'destroy':
            return [permissions.IsAdminUser()]
        return [permissions.IsAuthenticated()]

def custom_exception_handler(exc, context):
    """
    Consistent error shape for ALL API errors:
    {"success": false, "code": 404, "message": "...", "errors": {...}}
    """
    response = exception_handler(exc, context)   # let DRF handle first

        logger.warning('api_error', extra={

            'exc':    exc.__class__.__name__,
            'path':   context['request'].path,

            'code':    response.status_code,
            'message': str(exc.detail) if hasattr(exc, 'detail') else str(exc),
            'errors':  response.data,
        }
    return response

Part D — Guest Access

A Guest has no account. In Django REST Framework, guest access means AllowAny permission on public endpoints. No model is needed — the role is expressed in the API permission class, not the database.

    def has_permission(self, request, view):

    def has_permission(self, request, view):

# In a ViewSet — combine permissions per action
class StudentViewSet(viewsets.ModelViewSet):
    def get_permissions(self):
        if self.action in ['list', 'retrieve']:
            # admins, teachers, parents can view students
            return [(IsAdminRole | IsTeacherRole | IsParentRole)()]
        if self.action in ['create', 'update', 'destroy']:
            return [IsAdminRole()]      # only admins can modify students
        return [IsAuthenticated()]      # default: any logged-in user

class StudentViewSet(viewsets.ModelViewSet):

            qs = qs.filter(classroom__class_teacher__user=user)

            qs = qs.filter(user=user)

            qs = qs.filter(guardian__email=user.email)

        return StudentListSerializer if self.action == 'list' else StudentDetailSerializer

        if self.action in ['list', 'retrieve']:          return [permissions.IsAuthenticated()]
        if self.action in ['create','update','partial_update']: return [IsTeacherOrAdmin()]
        if self.action == 'destroy':                     return [permissions.IsAdminUser()]

        return Response({
            'student': StudentListSerializer(student, context={'request': request}).data,
            'marks':   MarkSerializer(marks, many=True).data,
            'gpa':     float(student.gpa),
        })

    @action(detail=True, methods=['get'], url_path='attendance-summary')
    def attendance_summary(self, request, pk=None):
        """GET /students/{pk}/attendance-summary/"""
        student = self.get_object()
        from django.db.models import Count
        summary = student.attendance.values('status').annotate(count=Count('id'))
        total   = student.attendance.count()
        present = student.attendance.filter(status='P').count()
        pct     = round((present / total * 100), 1) if total else 0
        return Response({'total': total, 'present': present,
                         'attendance_pct': pct, 'by_status': list(summary)})

from django.contrib.auth import get_user_model
from students.models import Student
from classes.models import Classroom

class UserFactory(factory.django.DjangoModelFactory):
    class Meta: model = get_user_model()
    email     = factory.Sequence(lambda n: ff"user{n}@school.com")
    full_name = factory.Faker('name')
    role      = 'student'
    password  = factory.PostGenerationMethodCall('set_password', 'testpass123')

class ClassroomFactory(factory.django.DjangoModelFactory):
    class Meta: model = Classroom
    name  = factory.Sequence(lambda n: ff"10-{chr(65+n)}")
    grade = 10;  section = factory.Sequence(lambda n: chr(65+n))

class StudentFactory(factory.django.DjangoModelFactory):
    class Meta: model = Student
    user      = factory.SubFactory(UserFactory)
    roll_no   = factory.Sequence(lambda n: ff"2024{n:03d}")
    classroom = factory.SubFactory(ClassroomFactory)
    dob       = factory.Faker('date_of_birth', minimum_age=12, maximum_age=18)
    gender    = 'M'

import pytest
from rest_framework.test import APIClient
from rest_framework import status
from .factories import UserFactory, StudentFactory, ClassroomFactory

    c = APIClient(); c.force_authenticate(user=teacher); return c


@pytest.mark.django_db
class TestStudentEndpoints:

    def test_list_unauthenticated_is_401(self):
        r = APIClient().get('/api/students/')
        assert r.status_code == status.HTTP_401_UNAUTHORIZED

    def test_list_returns_paginated(self, admin_client, db):
        StudentFactory.create_batch(5)
        r = admin_client.get('/api/students/')
        assert r.status_code == status.HTTP_200_OK
        assert 'pagination' in r.data
        assert r.data['pagination']['total'] == 5

    def test_create_student(self, admin_client, db):
        cls = ClassroomFactory()
        user = UserFactory(role='student')
        payload = {'roll_no':'2024099', 'user': user.id, 'classroom': cls.id,
                   'dob':'2010-06-15', 'gender':'M'}
        r = admin_client.post('/api/students/', payload, format='json')
        assert r.status_code == status.HTTP_201_CREATED
        assert r.data['roll_no'] == '2024099'

    def test_teacher_cannot_delete(self, teacher_client, db):
        s = StudentFactory()
        r = teacher_client.delete(ff'/api/students/{s.pk}/')
        assert r.status_code == status.HTTP_403_FORBIDDEN

    def test_invalid_roll_no_rejected(self, admin_client, db):
        r = admin_client.post('/api/students/', {'roll_no': 'ABCD001'}, format='json')
        assert r.status_code == status.HTTP_400_BAD_REQUEST
        assert 'roll_no' in r.data['errors']

    def test_classroom_capacity_rejected(self, admin_client, db):
        cls = ClassroomFactory(max_capacity=1)
        StudentFactory(classroom=cls)      # fill the 1-seat classroom
        user = UserFactory(role='student')
        r = admin_client.post('/api/students/',
                              {'roll_no':'2024098','user':user.id,'classroom':cls.id,
                               'dob':'2010-01-01','gender':'F'}, format='json')
        assert r.status_code == 409
        assert r.data['code'] == 'classroom_full'

    def test_filter_by_grade(self, admin_client, db):
        StudentFactory(classroom=ClassroomFactory(grade=10))
        StudentFactory(classroom=ClassroomFactory(grade=9))
        r = admin_client.get('/api/students/?grade=10')
        assert r.data['pagination']['total'] == 1

    def test_report_card_endpoint(self, admin_client, db):
        s = StudentFactory()
        r = admin_client.get(ff'/api/students/{s.pk}/report-card/')
        assert r.status_code == status.HTTP_200_OK
        assert 'student' in r.data and 'marks' in r.data

    def test_jwt_flow(self, db):
        user = UserFactory(role='admin', is_staff=True)
        c    = APIClient()
        r    = c.post('/api/auth/login/', {'email': user.email, 'password': 'testpass123'})
        assert r.status_code == 200
        assert 'access' in r.data and 'refresh' in r.data
        c.credentials(HTTP_AUTHORIZATION=ff"Bearer {r.data['access']}")
        r2 = c.get('/api/students/')
        assert r2.status_code == 200
